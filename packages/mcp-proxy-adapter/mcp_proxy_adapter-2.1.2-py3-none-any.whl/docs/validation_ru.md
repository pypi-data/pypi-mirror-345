```python
    if not is_valid:

        # –í —Å—Ç—Ä–æ–≥–æ–º —Ä–µ–∂–∏–º–µ –ø—Ä–µ—Ä—ã–≤–∞–µ–º —Ä–µ–≥–∏—Å—Ç—Ä–∞—Ü–∏—é
        if self.strict and not self.auto_fix:

```

### 1. DocstringValidator

–ü—Ä–æ–≤–µ—Ä—è–µ—Ç —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–æ–≤ —Ñ–æ—Ä–º–∞–ª—å–Ω—ã–º –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º —Ñ—É–Ω–∫—Ü–∏–∏:

- –ù–∞–ª–∏—á–∏–µ –æ–ø–∏—Å–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏
- –û–ø–∏—Å–∞–Ω–∏–µ –≤—Å–µ—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –≤ —Å–µ–∫—Ü–∏–∏ Args

- –°–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –∏–º–µ–Ω –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –≤ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–µ –∏ —Å–∏–≥–Ω–∞—Ç—É—Ä–µ —Ñ—É–Ω–∫—Ü–∏–∏

```python
class DocstringValidator:
    def validate(self, handler: Callable, command_name: str, metadata: Dict[str, Any]) -> Tuple[bool, List[str]]:
        errors = []
        
        # –ü–æ–ª—É—á–∞–µ–º —Ñ–æ—Ä–º–∞–ª—å–Ω—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã —Ñ—É–Ω–∫—Ü–∏–∏
        sig = inspect.signature(handler)
        formal_params = list(sig.parameters.keys())
        
        # –ü–∞—Ä—Å–∏–º –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥
        docstring = handler.__doc__ or ""
        parsed_doc = docstring_parser.parse(docstring)
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ –æ–ø–∏—Å–∞–Ω–∏—è —Ñ—É–Ω–∫—Ü–∏–∏
        if not parsed_doc.short_description and not parsed_doc.long_description:
            errors.append(f"–û—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç –æ–ø–∏—Å–∞–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏")
        
        # –ü–æ–ª—É—á–∞–µ–º –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –∏–∑ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–∞
        doc_params = {param.arg_name: param for param in parsed_doc.params}
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –≤—Å–µ —Ñ–æ—Ä–º–∞–ª—å–Ω—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –æ–ø–∏—Å–∞–Ω—ã –≤ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–µ
        for param in formal_params:
            if param not in doc_params and param != 'params':
                errors.append(f"–ü–∞—Ä–∞–º–µ—Ç—Ä '{param}' –Ω–µ –æ–ø–∏—Å–∞–Ω –≤ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–µ —Ñ—É–Ω–∫—Ü–∏–∏")
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ returns –≤ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–µ
        if not parsed_doc.returns:
            errors.append(f"–û—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç –æ–ø–∏—Å–∞–Ω–∏–µ –≤–æ–∑–≤—Ä–∞—â–∞–µ–º–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è –≤ –¥–æ–∫—Å—Ç—Ä–∏–Ω–≥–µ —Ñ—É–Ω–∫—Ü–∏–∏")
        
        return len(errors) == 0, errors
```

### 2. MetadataValidator

–ü—Ä–æ–≤–µ—Ä—è–µ—Ç —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –∫–æ–º–∞–Ω–¥—ã –µ—ë —Å–∏–≥–Ω–∞—Ç—É—Ä–µ:

- –û—Ç—Å—É—Ç—Å—Ç–≤–∏–µ –ª–∏—à–Ω–∏—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –≤ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö

```python
class MetadataValidator:
    def validate(self, handler: Callable, command_name: str, metadata: Dict[str, Any]) -> Tuple[bool, List[str]]:
        errors = []
        
        # –ü–æ–ª—É—á–∞–µ–º —Ñ–æ—Ä–º–∞–ª—å–Ω—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã —Ñ—É–Ω–∫—Ü–∏–∏
        sig = inspect.signature(handler)
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–∏–µ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö –∏ —Ñ–æ—Ä–º–∞–ª—å–Ω—ã—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤
        if "parameters" in metadata:
            for param_name, param_info in metadata["parameters"].items():
                # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ –ø–∞—Ä–∞–º–µ—Ç—Ä–∞ –≤ —Å–∏–≥–Ω–∞—Ç—É—Ä–µ —Ñ—É–Ω–∫—Ü–∏–∏
                if param_name not in sig.parameters:
                    errors.append(f"–ü–∞—Ä–∞–º–µ—Ç—Ä '{param_name}' —É–∫–∞–∑–∞–Ω –≤ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö, –Ω–æ –æ—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç –≤ —Å–∏–≥–Ω–∞—Ç—É—Ä–µ —Ñ—É–Ω–∫—Ü–∏–∏")
            
            # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –≤—Å–µ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã —É–∫–∞–∑–∞–Ω—ã –≤ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö
            for param_name, param in sig.parameters.items():
                if param.default == inspect.Parameter.empty and param_name != 'self':
                    if param_name not in metadata["parameters"]:
                        errors.append(f"–û–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä '{param_name}' –Ω–µ —É–∫–∞–∑–∞–Ω –≤ –º–µ—Ç–∞–¥–∞–Ω–Ω—ã—Ö")
        
        return len(errors) == 0, errors
```
